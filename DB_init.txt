CREATE TABLE FILM(
IDFILM INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
TITOLO VARCHAR(100),
TRAMA VARCHAR(100),
TRAILER VARCHAR(30),
GENERE VARCHAR(30));

CREATE TABLE CLIENTEREGISTRATO(
IDCLIENTE INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
NOME VARCHAR(30),
PASSWORD VARCHAR(30),
EMAIL VARCHAR(30),
CARTADICREDITO VARCHAR(30));

CREATE TABLE SALA(
IDSALA INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
CAPIENZA INT);

CREATE TABLE PROIEZIONE(
IDPROIEZIONE INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
DATA DATE,
ORARIO TIME,
IDFILM INT,
IDSALA INT,
FOREIGN KEY (IDFILM) REFERENCES FILM(IDFILM),
FOREIGN KEY (IDSALA) REFERENCES SALA(IDSALA));

CREATE TABLE BIGLIETTO(
IDBIGLIETTO INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
COSTO FLOAT,
NUMPOSTO INT,
IDCLIENTE INT,
IDPROIEZIONE INT,
FOREIGN KEY (IDCLIENTE) REFERENCES CLIENTEREGISTRATO(IDCLIENTE),
FOREIGN KEY (IDPROIEZIONE) REFERENCES PROIEZIONE(IDPROIEZIONE));
